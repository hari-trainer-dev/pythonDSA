Stacks & Queues

Implementation

Stacks follow LIFO (Last In, First Out); queues follow FIFO (First In, First Out). In Python, both can be implemented using lists or collections.deque for efficiency.

Without modules (using list):

# Stack
stack = []
stack.append(1)  # Push
stack.append(2)
print(stack.pop())  # Pop: 2
print(stack[-1])    # Peek: 1

# Queue
queue = []
queue.append(1)  # Enqueue
queue.append(2)
print(queue.pop(0))  # Dequeue: 1
print(queue[0])      # Front: 2

from collections import deque

# Stack (use append/pop for efficiency)
stack = deque()
stack.append(1)  # Push
stack.append(2)
print(stack.pop())  # Pop: 2
print(stack[-1])    # Peek: 1

# Queue (efficient pops from left)
queue = deque()
queue.append(1)  # Enqueue
queue.append(2)
print(queue.popleft())  # Dequeue: 1
print(queue[0])         # Front: 2

1. Using list 

class StackList:
    def __init__(self):
        self.items = []
    
    def push(self, x):
        self.items.append(x)
    
    def pop(self):
        if not self.is_empty():
            return self.items.pop()
        return None
    
    def peek(self):
        if not self.is_empty():
            return self.items[-1]
        return None
    
    def is_empty(self):
        return len(self.items) == 0

2. Using collections.deque 

from collections import deque

class StackDeque:
    def __init__(self):
        self.items = deque()
    
    def push(self, x):
        self.items.append(x)          # or appendleft()
    
    def pop(self):
        if not self.is_empty():
            return self.items.pop()   # or popleft()
        return None
    
    def peek(self):
        if not self.is_empty():
            return self.items[-1]     # or self.items[0]
        return None
    
    def is_empty(self):
        return len(self.items) == 0


# Quick usage example
s = StackList()
s.push(10); s.push(20); s.push(30)
print(s.pop())   # 30
print(s.peek())  # 20

Monotonic Stacks

A stack maintaining elements in strictly increasing or decreasing order by popping elements that violate the order when adding new ones. Useful for finding next/previous greater/smaller elements.

Without modules (using list):

# Monotonic increasing stack (bottom to top: small to large)
def monotonic_increasing(arr):
    stack = []
    for num in arr:
        while stack and stack[-1] >= num:  # Pop until valid
            stack.pop()
        stack.append(num)
    return stack

print(monotonic_increasing([4, 2, 5, 3, 1]))  # [1] or similar based on order

With modules (using collections.deque):

from collections import deque

# Monotonic decreasing stack (bottom to top: large to small)
def monotonic_decreasing(arr):
    stack = deque()
    for num in arr:
        while stack and stack[-1] <= num:  # Pop until valid
            stack.pop()
        stack.append(num)
    return list(stack)

print(monotonic_decreasing([4, 2, 5, 3, 1]))  # [5, 3, 1]

Next Greater Element

For each element in an array, find the next greater element to its right (or -1 if none). Solved using a monotonic decreasing stack to track candidates.

Without modules (using list):

def next_greater_element(nums):
    n = len(nums)
    result = [-1] * n
    stack = []  # Monotonic decreasing
    for i in range(n-1, -1, -1):  # Reverse traversal
        while stack and stack[-1] <= nums[i]:
            stack.pop()
        if stack:
            result[i] = stack[-1]
        stack.append(nums[i])
    return result

print(next_greater_element([4, 5, 2, 25]))  # [5, 25, 25, -1]

1. Valid Parentheses

Check if a string containing only ( ) { } [ ] is valid (properly nested and closed).

Key idea: Use stack → push opening, pop on closing and match.

def isValid(s: str) -> bool:
    stack = []
    mapping = {')':'(', '}':'{', ']':'['}

    for char in s:
        if char in mapping:           # closing bracket
            if not stack or stack.pop() != mapping[char]:
                return False
        else:                         # opening bracket
            stack.append(char)

    return len(stack) == 0


# Examples
print(isValid("()[]{}"))       # True
print(isValid("(]"))           # False
print(isValid("({[]})"))       # True
print(isValid("([)]"))         # False
print(isValid(""))             # True

Most frequent follow-up:

→ What if there are letters/numbers too? → just ignore non-brackets or return False depending on problem statement.

2. Queue using Stacks (Implement Queue using two Stacks)

Two most common variants asked:

A. Amortized O(1) (most common in interviews)

class MyQueue:
    def __init__(self):
        self.in_stack  = []    # for push
        self.out_stack = []    # for pop/peek

    def push(self, x: int) -> None:
        self.in_stack.append(x)

    def pop(self) -> int:
        self._move_if_needed()
        return self.out_stack.pop()

    def peek(self) -> int:
        self._move_if_needed()
        return self.out_stack[-1]

    def empty(self) -> bool:
        return not self.in_stack and not self.out_stack

    def _move_if_needed(self):
        if not self.out_stack:
            while self.in_stack:
                self.out_stack.append(self.in_stack.pop())

Time complexity:

  • push → O(1)
  • pop / peek → amortized O(1), worst-case O(n)

B. Every operation O(n) (less common, but sometimes asked)

Just use one stack and reverse twice.

Dequeue Applications

Most important one (1. Sliding Window Maximum) 

from collections import deque

def maxSlidingWindow(nums: list[int], k: int) -> list[int]:
    if not nums:
        return []

    deq = deque()           # stores indices (decreasing order of values)
    result = []

    for i in range(len(nums)):
        # 1. remove elements smaller than current from back
        while deq and nums[deq[-1]] < nums[i]:
            deq.pop()

        deq.append(i)

        # 2. remove out of window
        if deq[0] == i - k:
            deq.popleft()

        # 3. after window size reached → add max
        if i >= k - 1:
            result.append(nums[deq[0]])

    return result


print(maxSlidingWindow([1,3,-1,-3,5,3,6,7], 3))
# [3,3,5,5,6,7]

2. Generate Binary Numbers from 1 to n

from collections import deque

def generate_binary_numbers(n):
    if n <= 0:
        return []
    
    result = []
    q = deque()
    q.append("1")               # start with 1
    
    while len(result) < n:
        current = q.popleft()
        result.append(current)
        
        q.append(current + "0")     # add 0 at end
        q.append(current + "1")     # add 1 at end
    
    return result


# Test
print(generate_binary_numbers(10))
# ['1', '10', '11', '100', '101', '110', '111', '1000', '1001', '1010']
